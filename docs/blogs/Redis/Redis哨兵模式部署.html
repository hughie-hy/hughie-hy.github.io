    # 编写yml文件

    version: '3.9'
 
networks:
  app-server:
    external: true
 
services:
  redis-server1:
    image: nexus.ubssdic.com:8445/redis:7.0.10
    volumes:
      - /data/redis/server1/redis.conf:/usr/local/etc/redis/redis.conf
    ports:
      - 6371:6379
    networks:
      - app-server
    deploy:
      placement:
        constraints:
          - "node.hostname == app141"
    command: ["redis-server", "/usr/local/etc/redis/redis.conf"]
 
  redis-server2:
    image: nexus.ubssdic.com:8445/redis:7.0.10
    volumes:
      - /data/redis/server2/redis.conf:/usr/local/etc/redis/redis.conf
    ports:
      - 6372:6379
    networks:
      - app-server
    depends_on:
      - redis-server1
    deploy:
      placement:
        constraints:
          - "node.hostname == app141"
    command: ["redis-server", "/usr/local/etc/redis/redis.conf"]
 
  redis-server3:
    image: nexus.ubssdic.com:8445/redis:7.0.10
    volumes:
      - /data/redis/server3/redis.conf:/usr/local/etc/redis/redis.conf
    ports:
      - 6373:6379
    networks:
      - app-server
    depends_on:
      - redis-server1
    deploy:
      placement:
        constraints:
          - "node.hostname == app141"
    command: ["redis-server", "/usr/local/etc/redis/redis.conf"]
 
  redis-sentinel1:
    image: nexus.ubssdic.com:8445/redis:7.0.10
    volumes:
      - /data/redis/sentinel1/sentinel.conf:/usr/local/etc/redis/sentinel.conf
    ports:
      - 26371:6379
    networks:
      - app-server
    depends_on:
      - redis-server1
      - redis-server2
      - redis-server3
    deploy:
      placement:
        constraints:
          - "node.hostname == app141"
    command: ["redis-sentinel", "/usr/local/etc/redis/sentinel.conf"]
 
  redis-sentinel2:
    image: nexus.ubssdic.com:8445/redis:7.0.10
    volumes:

    docker stack rm redis-sentinel
