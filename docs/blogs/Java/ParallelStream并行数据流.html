    Stream流

    hughie.cheng 10/10/2023 Java基础

    一般数据流与并行数据流对比

    public class ThreadTest {
    public List<FundInfo> fundList = new ArrayList<>() {{
        add(new FundInfo("2002", "瑞银"));
        add(new FundInfo("2042", "瑞源"));
        add(new FundInfo("1302", "瑞信"));
        add(new FundInfo("1002", "白银"));
        add(new FundInfo("4002", "黄金"));
    }};
    

    class FundInfo {
        
        public FundInfo(String fundCode, String fundName) {
            this.fundCode = fundCode;
            this.fundName = fundName;
        }

        String fundCode;
        String fundName;

        public String getFundName() {
            return fundName;
        }

        public void setFundName(String fundName) {
            this.fundName = fundName;
        }

        public FundInfo process() {
            this.setFundName(this.getFundName() + ":processed");
            try {
                Thread.sleep(2000);
            } catch (InterruptedException e) {
                throw new RuntimeException(e);
            }
            return this;
        }
    }

    void init() {

    }

    @Test
    void parallel() {
        StopWatch stopWatch = new StopWatch("watchDog");
        stopWatch.start();
        //未使用并行数据流
        fundList.stream().map(FundInfo::process).collect(Collectors.toList());
        stopWatch.stop();

        stopWatch.start();
        //使用并行数据流
        fundList.parallelStream().map(FundInfo::process).collect(Collectors.toList());
        stopWatch.stop();
        System.out.println(stopWatch.prettyPrint(TimeUnit.SECONDS));
    }
}
