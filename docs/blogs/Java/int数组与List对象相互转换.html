    # int[]转List

    手动将数组元素逐个放入list中，进行转换

    /**
 * 手动将数组元素逐个放入list中，进行转换
 */
public List<Integer> manual2List(int[] arrays) {
    List<Integer> list = new ArrayList<>();
    for (int array : arrays) {
        list.add(array);
    }
    return list;
}

    # List转int[]

    /**
 * 手动创建数组，将list中的元素逐个添加到int[]数组中
 */
public int[] manualList2Array(List<Integer> list) {
    int[] res = new int[list.size()];
    int i = 0;
    for (int num : list) {
        res[i++] = num;
    }
    return res;
}
